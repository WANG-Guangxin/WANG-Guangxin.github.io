function [Ld,Lb] = fen_qu(Lin)
i = 1;
j = 1;
[m,n] = size(Lin);
Ld = 0;
Lb = 0;
for k = 1:m*n
    x = Lin(k);
    if x <= 0.5
        Ld(i) = x;
        i = i + 1;
    end
    if x > 0.5
        Lb(j) = x;
        j = j + 1;
    end
end
end
