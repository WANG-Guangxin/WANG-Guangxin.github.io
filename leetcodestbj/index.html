            if parent[index] != index:
                parent[index] = find(parent[index])
            return parent[index]
        
        def union(index1: int, index2: int):
            parent[find(index1)] = find(index2)

        for node1, node2 in edges:

                union(node1, node2)
            else:
                return [node1, node2]
        
        return []

    def find(self, x: int) -> int:

            return x
        if self.f[x] == x:
